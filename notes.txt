CREATE TABLE "urls" (
	"id"	INTEGER,
	"urls"	TEXT,
	"titles"	TEXT,
	"lib"	TEXT,
	"libtag"	TEXT,
	"css"	TEXT,
	"csstag"	TEXT,
	"filename"	TEXT,
	PRIMARY KEY("id" AUTOINCREMENT)
);

	"id"	INTEGER,

);
